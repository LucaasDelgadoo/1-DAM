CREATE TRIGGER VERIFICA_REPVENTAS
ON REPVENTAS
AFTER INSERT
AS
BEGIN

DECLARE @EMPLEADODERECURSOSHUMANOS VARCHAR(50), @FECHA DATE, @NOMBRETRABAJADOR VARCHAR(50);

SET @EMPLEADODERECURSOSHUMANOS = SYSTEM_USER;
SET @FECHA = GETDATE();
SELECT NOMBRE = @NOMBRETRABAJADOR FROM INSERTED;

INSERT INTO historico VALUES (@EMPLEADODERECURSOSHUMANOS, @FECHA, @NOMBRETRABAJADOR);

END;
